    <script>

        // Get all steps and circles

        const steps = document.querySelectorAll('.step');

        const circles = document.querySelectorAll('.circle');


        // Define what happens at each step

        const animations = {

            0: () => {

                // Initial state - just center circle

                circles[0].setAttribute('r', '50');

                circles[0].setAttribute('opacity', '0.9');

                circles[1].setAttribute('opacity', '0');

                circles[2].setAttribute('opacity', '0');

                circles[3].setAttribute('opacity', '0');

                circles[4].setAttribute('opacity', '0');

            },

            1: () => {

                // Show surrounding circles

                circles[0].setAttribute('r', '40');

                circles[1].setAttribute('opacity', '0.8');

                circles[2].setAttribute('opacity', '0.8');

                circles[3].setAttribute('opacity', '0.8');

                circles[4].setAttribute('opacity', '0.8');

            },

            2: () => {

                // Grow all circles

                circles[0].setAttribute('r', '60');

                circles[1].setAttribute('r', '50');

                circles[2].setAttribute('r', '50');

                circles[3].setAttribute('r', '50');

                circles[4].setAttribute('r', '50');

            },

            3: () => {

                // Spread out circles

                circles[1].setAttribute('cx', '100');

                circles[2].setAttribute('cx', '400');

                circles[3].setAttribute('cy', '100');

                circles[4].setAttribute('cy', '400');

            },

            4: () => {

                // Final state - all equal size

                circles[0].setAttribute('r', '50');

                circles[1].setAttribute('r', '50');

                circles[2].setAttribute('r', '50');

                circles[3].setAttribute('r', '50');

                circles[4].setAttribute('r', '50');

                circles[1].setAttribute('cx', '150');

                circles[2].setAttribute('cx', '350');

                circles[3].setAttribute('cy', '150');

                circles[4].setAttribute('cy', '350');

            }

        };


        // Intersection Observer to detect which step is active

        const observerOptions = {

            root: null,

            rootMargin: '-40% 0px -40% 0px',

            threshold: 0

        };


        const observer = new IntersectionObserver((entries) => {

            entries.forEach(entry => {

                if (entry.isIntersecting) {

                    // Remove active class from all steps

                    steps.forEach(s => s.classList.remove('active'));

                    

                    // Add active class to current step

                    entry.target.classList.add('active');

                    

                    // Get step number and run animation

                    const stepNum = entry.target.dataset.step;

                    if (animations[stepNum]) {

                        animations[stepNum]();

                    }

                }

            });

        }, observerOptions);


        // Observe all steps

        steps.forEach(step => observer.observe(step));


        // Initialize with first animation

        animations[0]();

    </script>
